function buildTabs() {
  const strip = document.getElementById('tabStrip');
  strip.innerHTML = TABS.map(t =>
    `<button class="tab-btn ${t.id===activeTab?'active':''}" data-tab="${t.id}">${t.label}</button>`
  ).join('');
  strip.querySelectorAll('.tab-btn').forEach(btn => {
    btn.onclick = () => { activeTab = btn.dataset.tab; buildTabs(); renderContent(); };
  });
}

}

// ═══ INVENTORY ═══
function buildInventory() {
  const grid = document.getElementById('invGrid');
  let html = '';
  for (let i = 0; i < 30; i++) {
    const filled = Math.random() > 0.7;
    html += `<div class="inv-cell ${filled?'filled':''}"></div>`;
  }
  grid.innerHTML = html;
}

// ═══ HOTBAR ═══
function buildHotbar() {
  const bar = document.getElementById('hotbar');
  const keys = ['1','2','3','4','5','6','7','8','9','0'];
  bar.innerHTML = keys.map(k => `<div class="hb-slot"><span class="key">${k}</span></div>`).join('');
}
